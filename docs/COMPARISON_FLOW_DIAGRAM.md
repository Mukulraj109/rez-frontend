## ğŸ”§ Hook Usage Patterns

### Pattern 1: Full Access (Product Card)

```typescript
import { useComparison } from '@/contexts/ComparisonContext';

function ProductCard({ product }) {
  const {
    addProduct,
    removeProduct,
    isInComparison,
    canAddMore
  } = useComparison();

  const inComparison = isInComparison(product.id);

  return (
    <Button
      title={inComparison ? "Remove" : "Compare"}
      onPress={() =>
        inComparison
          ? removeProduct(product.id)
          : addProduct(product)
      }
      disabled={!inComparison && !canAddMore}
    />
  );
}
```

### Pattern 2: Actions Only (Add Button)

```typescript
import { useComparisonActions } from '@/contexts/ComparisonContext';

function AddToCompareButton({ product }) {
  const { addProduct } = useComparisonActions();

  return (
    <Button
      title="Compare"
      onPress={() => addProduct(product)}
    />
  );
}
```

### Pattern 3: Status Only (Badge)

```typescript
import { useComparisonStatus } from '@/contexts/ComparisonContext';

function HeaderBadge() {
  const { count } = useComparisonStatus();

  if (count === 0) return null;

  return (
    <View style={styles.badge}>
      <Text>{count}</Text>
    </View>
  );
}
```
